12:21:53 <p>
12:21:53 The TclHttpd is a pure Tcl implementation of a Web server.
12:21:53 It runs as a Tcl script on top of
12:21:53 <a href=http://www.tcl.tk/software/tcltk/>Tcl</a>.
12:21:53 This server is running Tcl 8.4.8.
12:21:53 There is more <a href=version_history.html>TclHttpd version information</a>
12:21:53 about using other versions of Tcl and the Standard Tcl Library (tcllib).
12:21:53 <p>
12:21:53 
12:21:53 While this server works fine as a stand alone web server, the
12:21:53 intent is to embed the server in other applications in order to
12:21:53 "web enable" them.
12:21:53 A Tcl-based web server is ideal for embedding
12:21:53 because Tcl was designed from the start
12:21:53 to support embedding into other applications.
12:21:53 The interpreted nature of Tcl allows dynamic reconfiguration of the
12:21:53 server. Once the core interface between the web server and the hosting
12:21:53 application is defined, it is possible to manage the web server, upload
12:21:53 Safe-Tcl control scripts, download logging information, and otherwise
12:21:53 debug the Tcl part of the application without restarting the hosting
12:21:53 application.
12:21:53 <p>
12:21:53 
12:21:53   The server is distributed under a <a href=license.terms>copyright</a> that allows free use.
12:21:53 
12:21:53 
12:21:53 <p>
12:21:53 The <a href="mailto:tclhttpd-users@lists.sourceforge.net">tclhttpd-users@lists.sourceforge.net</a> mailing list is the best source of current
12:21:53 information about TclHttpd.

12:24:01 <p>
12:24:01 The TclHttpd is a pure Tcl implementation of a Web server.
12:24:01 It runs as a Tcl script on top of
12:24:01 <a href=http://www.tcl.tk/software/tcltk/>Tcl</a>.
12:24:01 This server is running Tcl 8.4.8.
12:24:01 There is more <a href=version_history.html>TclHttpd version information</a>
12:24:01 about using other versions of Tcl and the Standard Tcl Library (tcllib).
12:24:01 <p>
12:24:01 
12:24:01 While this server works fine as a stand alone web server, the
12:24:01 intent is to embed the server in other applications in order to
12:24:01 "web enable" them.
12:24:01 A Tcl-based web server is ideal for embedding
12:24:01 because Tcl was designed from the start
12:24:01 to support embedding into other applications.
12:24:01 The interpreted nature of Tcl allows dynamic reconfiguration of the
12:24:01 server. Once the core interface between the web server and the hosting
12:24:01 application is defined, it is possible to manage the web server, upload
12:24:01 Safe-Tcl control scripts, download logging information, and otherwise
12:24:01 debug the Tcl part of the application without restarting the hosting
12:24:01 application.
12:24:01 <p>
12:24:01 
12:24:01   The server is distributed under a <a href=license.terms>copyright</a> that allows free use.
12:24:01 
12:24:01 
12:24:01 <p>
12:24:01 The <a href="mailto:tclhttpd-users@lists.sourceforge.net">tclhttpd-users@lists.sourceforge.net</a> mailing list is the best source of current
12:24:01 information about TclHttpd.

# opis prot SOCKS v.4
proc socks4connect {proxy_host proxy_port host port {username {}}} {
  set connect_request \x04\x01
  append connect_request [binary format S $port]
  if {[regexp {[0-9]+.[0-9]+.[0-9]+.[0-9]+} $host]} {
    set use_host false
    append connect_request [binary format c4 [split $host .]]
  } else {
    # ip address 0.0.0.x is always invalid and signals that a hostname is specified
    set use_host true
    append connect_request [binary format c4 [split 0.0.0.1 .]]
  }
  append connect_request $username
  append connect_request \x00
  if {$use_host} {
    append connect_request $host
    append connect_request \x00
  }
  set s [socket $proxy_host $proxy_port]
  fconfigure $s -translation binary -encoding binary
  puts -nonewline $s $connect_request; flush $s
  set response [read $s 8]
  set rr [string index $response 1]
  if {$rr ne "\x5a"} {
    error "connection request rejected by proxy ($rr)"
  } else {
    return $s
  }
}
  #%
