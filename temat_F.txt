  #% sock5
fileevent $sock readable "obsluga $sock"; # budowanie odpowiedzi na pytanie klienta...

proc obsluga s {
  set x [read $s]; set peer [fconfigure $s -peer]
  puts "$peer : $x"
  fconfigure $s -remote $peer
  puts -nonewline $s "$x (odp)"; flush $s
}                                                                                                               
  #%
